문제)
자연수 A, B가 주어지면 A부터 B까지의 합을 수식과 함께 출력하세요.

### 제출

``` Cpp
#include <iostream>
#define _CRT_SECURE_NO_WARNINGS

using namespace std;

int main() 
{
	//FILE* fp = nullptr;
	//freopen_s(&fp, "input.txt", "rt", stdin);

	int a, b, sum = 0;
	cin >> a >> b;

	for (int i = a; i < b; i++)
	{
		sum += i;
		cout << i << " +";;
	}
	cout << b << " = " << sum + b;

	return 0;
}
```

### 모범 답안

``` Cpp
#include<stdio.h>
int main(){
	int a, b, sum=0, i;
	scanf("%d %d", &a, &b);
	for(i=a; i<b; i++){
		sum+=i;
		printf("%d + ", i);
	}
	printf("%d = %d\n", i, sum+i);
	return 0;
}
```
